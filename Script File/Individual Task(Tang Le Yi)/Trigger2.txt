--Trigger2:Automatically Set Member Type to VIP
CREATE OR REPLACE TRIGGER trg_set_member_vip
AFTER INSERT ON vip_subscription
FOR EACH ROW
BEGIN
    UPDATE member
    SET member_type = 'VIP'
    WHERE member_id = :NEW.member_id;
END;
/
